Stability Plot

% Stability plot
alpha = -10:0.01:10;
beta = -10:0.01:10;
nreal = length(alpha);
nimag = length(beta);
absR = zeros(nreal,nimag);
% For ESDIRK23 :
gamma = 1-1/sqrt(2);
a31 = (1-gamma)/2;
AT = [0 gamma a31;0 gamma a31;0 0 gamma];
b = AT(:,3);
A = AT';
I = eye(size(A));
e = ones(size(A,1),1);
for kreal = 1:nreal
for kimag = 1:nimag
z = alpha(kreal)+i*beta(kimag);
tmp = (I-z*A)\e;
R = 1 + z*b'*tmp;
absR(kimag,kreal) = abs(R);
end
end
imagesc(alpha,beta,absR,[0,1]);
colorbar
grid on
axis image
axis xy
xlabel('real');
ylabel('imag');
title('ESDIRK23 Stability - |R(Z)|');
